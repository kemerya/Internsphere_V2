const express = require("express"); const bodyParser = require("body-parser"); const fs = require("fs"); const app = express(); app.use(bodyParser.json()); app.use(bodyParser.urlencoded({ extended: true })); // Mock storage (replace with database like MongoDB or MySQL for production) const dataFile = "reports.json"; // Load reports from the file function loadReports() { if (fs.existsSync(dataFile)) { return JSON.parse(fs.readFileSync(dataFile, "utf8")); } return []; } // Save reports to the file function saveReports(reports) { fs.writeFileSync(dataFile, JSON.stringify(reports, null, 2)); } // API to submit the form app.post("/submit-report", (req, res) => { const reports = loadReports(); const newReport = req.body; // Add the new report and save reports.push(newReport); saveReports(reports); res.json({ message: "Report submitted successfully!" }); }); // API to fetch reports app.get("/reports", (req, res) => { const reports = loadReports(); res.json(reports); }); // Serve static files (for the HTML pages) app.use(express.static("public")); // Start server const PORT = 3000; app.listen(PORT, () => { console.log(`Server running at http://localhost:${PORT}`); });
